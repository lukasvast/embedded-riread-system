342 void LCD_DrawSymbol(unsigned int x, unsigned int y, unsigned int w, unsigned int h, unsigned char t, const unsigned char *pImage)
343 {
344  unsigned char colorMsb;
345  unsigned char colorLsb;
346  unsigned char colorDup = 0;
347  unsigned int color=0;
348  unsigned int x1,y1;
349 
350  // set draw area
351  LCD_SetArea(x,y,x+w-1,y+h-1);
352  for (y1 = 0; y1<h ; y1++)
353  {
354  for (x1 = 0; x1<w ; x1++)
355  {
356  if (!colorDup)
357  {
358  colorMsb = pgm_read_byte(pImage++);
359  colorLsb = pgm_read_byte(pImage++);
360  if (t == 0x01)
361  {
362  colorDup = colorLsb & 0x20; // get duplicate flag
363  colorLsb = colorLsb | 0x20; // remove duplicate flag and add more white
364  }
365  if (colorDup)
366  colorDup = pgm_read_byte(pImage++); // get duplicate no
367  color = ((uint16_t)colorMsb << 8) | colorLsb; // merge color
368  SetColor(color); // set a pixel
369  }
370  else
371  {
372  colorDup--;
373  }
374  LCD_WrDat(color);
375  }
376  }
377  LCD_SetArea(0, 0, GetMaxX(), GetMaxY());
378 }
